class DracoChatViewModel : ViewModel() {
    private val _uiState = MutableStateFlow<ChatState>(ChatState.Idle)
    val uiState: StateFlow<ChatState> = _uiState.asStateFlow()

    fun sendMessage(prompt: String) = viewModelScope.launch {
        _uiState.value = ChatState.Loading
        val result = aiRepository.generateResponse(prompt)
        _uiState.value = ChatState.Success(result)
    }
}
